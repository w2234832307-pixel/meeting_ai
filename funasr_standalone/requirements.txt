# 安装说明：
# 
# 1. 基础安装（CPU/GPU 通用）：
#    conda activate meeting_ai_pyannote
#    pip install -r requirements.txt
#
# 2. GPU 版本（如果使用 CUDA 12.x，torch 已包含 CUDA 支持）：
#    直接安装即可，torch==2.8.0 已包含 CUDA 12.x 支持
#
# 3. 如果需要 CUDA 11.8：
#    pip install torch==2.8.0 torchaudio==2.8.0 --index-url https://download.pytorch.org/whl/cu118
#
# 4. Pyannote 首次使用注意事项：
#    - 需要在 HuggingFace 上接受模型使用协议
#      https://huggingface.co/pyannote/speaker-diarization-3.1
#    - 需要设置环境变量 HF_TOKEN（如果使用在线模型）：
#      export HF_TOKEN=your_huggingface_token
#    - 如果使用离线模式，设置：
#      export HF_HUB_OFFLINE=1
#      export TRANSFORMERS_OFFLINE=1
